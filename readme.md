```bash
# apt install python3-pip python3-gi phantomjs mitmproxy ffmpeg gir1.2-gtk-vnc-2.0 \
```

```bash
$ git clone https://github.com/r3vn/badKarma.git
```

```bash
# cd badKarma
# pip3 install -r requirements.txt
```
